Классы ядра и назначение

Класс Назначение
EngineSingleton движка: canvas, цикл игры, экран, камеры, сцены.
SceneКонтейнер игровых объектов.
GameObjectИгровая сущность с набором компонентов.
GameCollectionУниверсальная коллекция c каскадным lifecycle.
InputSystemКлавиатурный ввод (down/pressed/released).
CollisionSystemГлобальная проверка пересечений коллайдеров.
AssetLoaderЗагрузка изображений и аудио в Map.
AudioManagerКаналы звука, музыка, sfx, mute/pause/volume.
BaseComponentБаза для всех компонентов ECS.
EventSystemЗаготовка под future event bus.

Пример: сцена + объект + ввод

class Player extends GameObject {
  constructor() {
    super();
    this.speed = 300;
    this.bindComponent(new SpriteComponent({ width: 64, height: 64, color: "#f59e0b" }));
  }

  update(dt) {
    super.update(dt);
    const input = Engine.instance.input;
    if (input.isKeyDown("KeyA")) this.transform.translate(-this.speed * dt, 0);
    if (input.isKeyDown("KeyD")) this.transform.translate(this.speed * dt, 0);
  }
}

class MainScene extends Scene {
  init() {
    this.player = new Player();
    this.player.transform.setPosition(200, 200);
    this.gameObjects.append(this.player);
  }
}
