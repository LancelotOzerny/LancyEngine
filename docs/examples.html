Раздел 8: Практические примеры

Готовые шаблоны кода, которые можно брать за основу для реальных игровых сцен.

Главная Назад

Пример 1: Игрок с физикой и прыжком

class Player extends GameObject {
  constructor() {
    super();
    this.bindComponent(new SpriteComponent({
      width: 56,
      height: 56,
      color: "#facc15"
    }));

    this.bindComponent(new ColliderComponent({
      width: 48,
      height: 48,
      offsetX: 4,
      offsetY: 4,
      layer: 1,
      mask: [2]
    }));

    this.body = new PhysicComponent({
      gravityScale: 2.4,
      drag: 0.1
    });
    this.bindComponent(this.body);
  }

  update(dt) {
    super.update(dt);
    if (Engine.instance.input.isKeyPressed("Space")) {
      this.body.velocity.y = -520;
      Engine.instance.audio.playSfx("jump", { volume: 0.8 });
    }
  }
}

Пример 4: Sprite sheet + state machine

class Player extends GameObject {
  constructor() {
    super();
    this.anim = new AnimationComponent({
      sprite: "hero_sheet",
      frameWidth: 64,
      frameHeight: 64,
      initialState: "idle",
      states: {
        idle: {
          fps: 6,
          frames: [0, 1, 2, 3],
          transitions: [{ to: "run", param: "isMoving", equals: true }]
        },
        run: {
          fps: 12,
          frames: [8, 9, 10, 11, 12, 13],
          transitions: [{ to: "idle", param: "isMoving", equals: false }]
        }
      }
    });
    this.bindComponent(this.anim);
  }

  update(dt) {
    super.update(dt);
    const moving = Engine.instance.input.isKeyDown("KeyA") || Engine.instance.input.isKeyDown("KeyD");
    this.anim.setParam("isMoving", moving);
  }
}

Пример 5: Полный запуск с ассетами

Engine.instance.init("#game", {
  width: 1920,

  backgroundColor: "#0f1720",
  screenMode: "expand"
});

}, {

  audio: ["./assets/theme.mp3", "./assets/jump.wav", "./assets/hit.wav"]
